1) Select the employee in department 30.
answer: select * from employees where department_id='30';

2) List the names, numbers and department of all clerks.
answer: select first_name, last_name, phone_number, department_id from employees 
	where job_id in ('PU_CLERK', 'ST_CLERK', 'SH_CLERK');

3) Find the depart numbers and the name of employee of all dept with Deptno greater or equal to 20.
answer: select department_id, first_name, last_name from employees

answer: select * from employees where (commission_pct*salary)>salary;

14) Find the details of all the manager in dept 10, all clerk in dept 20  
answer:  select * from employees join jobs using(job_id)
  where ((lower(jobs.job_title) like '%manager' and  employees.department_id = 10 ) or (lower(jobs.job_title) like '%clerk' and employees.department_id = 20 ));


15) And all employees who are neither clerks nor manager but whose salary is greater than or equal to Rs. 2000.
answer: select * from employees join jobs using(job_id)
  	where not (lower(jobs.job_title) like '%manager' or lower(jobs.job_title) like '%clerk') and salary>= 2000;

 If all employees not receiving commission are entailed to Rs. 250, Show the net earnings of all employees.
